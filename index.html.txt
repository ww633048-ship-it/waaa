        // ========== إعدادات تسجيل الدخول ==========
        const ADMIN_USERNAME = "Waleed";      // غيّره لأي اسم تبيه
        const ADMIN_PASSWORD = "kvn2026";     // غيّره لكلمة سر قوية
        const AUTH_TOKEN = "waleed2026kvn";   // نفس التوكن في السيرفر

        const SERVER_URL = window.location.origin;
        let currentAiCode = "", currentAiScriptId = "", allKeysData = [], currentEditId = null;

        // ========== دوال مساعدة ==========
        function getHeaders() {
            const headers = { 'Content-Type': 'application/json' };
            const token = sessionStorage.getItem('auth_token');
            if (token) {
                headers['Authorization'] = 'Bearer ' + token;
            }
            return headers;
        }

        async function apiFetch(url, options = {}) {
            const res = await fetch(url, { ...options, headers: getHeaders() });
            if (!res.ok) {
                if (res.status === 403) {
                    alert("⛔ انتهت الجلسة، الرجاء تسجيل الدخول مجدداً");
                    logout();
                    throw new Error("غير مصرح");
                }
                console.warn("استجابة غير متوقعة:", res.status);
            }
            return res;
        }

        // ========== تسجيل الدخول / الخروج ==========
        function doLogin() {
            const user = document.getElementById('login-username').value.trim();
            const pass = document.getElementById('login-password').value.trim();
            if (user === ADMIN_USERNAME && pass === ADMIN_PASSWORD) {
                sessionStorage.setItem('waleed_logged_in', 'true');
                sessionStorage.setItem('auth_token', AUTH_TOKEN);
                document.getElementById('login-screen').classList.add('hidden');
                document.getElementById('admin-panel').classList.remove('hidden');
                initPanel();
            } else {
                document.getElementById('login-error').classList.remove('hidden');
            }
        }

        function logout() {
            sessionStorage.removeItem('waleed_logged_in');
            sessionStorage.removeItem('auth_token');
            location.reload();
        }

        // التحقق من الجلسة عند تحميل الصفحة
        if (sessionStorage.getItem('waleed_logged_in') === 'true') {
            if (!sessionStorage.getItem('auth_token')) {
                sessionStorage.setItem('auth_token', AUTH_TOKEN);
            }
            document.getElementById('login-screen').classList.add('hidden');
            document.getElementById('admin-panel').classList.remove('hidden');
            initPanel();
        }

        async function refreshKeys() { await loadMyKeys(); }

        async function uploadScript(overrideSource = null, overrideId = null) {
            const id = overrideId || document.getElementById('script-id').value.trim();
            let source = overrideSource || document.getElementById('source-code').value.trim();
            const type = document.querySelector('input[name="script-type"]:checked').value;
            if (!id || !source) return alert("⚠️ املأ الحقول!");
            const res = await apiFetch(`${SERVER_URL}/api/upload`, {
                method: 'POST',
                body: JSON.stringify({ id, source, type })
            });
            const data = await res.json();
            if (data.success) {
                let ls = `loadstring(game:HttpGet("${SERVER_URL}/load/${id}.lua"))()`;
                if (type === 'paid') ls = `_G.Key = "مفتاحك"\n` + ls;
                document.getElementById('final-link-input').value = ls;
                document.getElementById('result-box').classList.remove('hidden');
                loadMyScripts(); loadMyKeys();
                alert("✅ تم الحفظ بنجاح!");
            } else alert("❌ " + data.message);
        }

        async function uploadWithKeySystem() {
            const id = document.getElementById('script-id').value.trim();
            let source = document.getElementById('source-code').value.trim();
            if (!id || !source) return alert("⚠️ املأ الحقول!");
            document.querySelector('input[name="script-type"][value="paid"]').checked = true;
            await uploadScript(source, id);
        }

        async function injectServerBinding() {
            const id = document.getElementById('script-id').value.trim();
            let source = document.getElementById('source-code').value.trim();
            if (!id || !source) return alert("⚠️ اكتب اسم الملف وضع الكود أولاً!");
            const bindingCode = `-- KVN Server Binding
local Players = game:GetService("Players")
local StarterGui = game:GetService("StarterGui")
local HttpService = game:GetService("HttpService")
local SERVER_URL = "${SERVER_URL}"
local SCRIPT_ID = "${id}"
local localPlayer = Players.LocalPlayer

local function checkStatus()
    local success, response = pcall(function() return game:HttpGet(SERVER_URL .. "/api/script-status/" .. SCRIPT_ID) end)
    if not success then return false, "فشل الاتصال" end
    local data = HttpService:JSONDecode(response)
    if not data.active then return false, data.message or "⛔ متوقف" end
    return true
end

local function kick(msg)
    StarterGui:SetCore("SendNotification", { Title = "Waleed Hub", Text = msg, Duration = 5 })
    wait(1)
    localPlayer:Kick(msg)
end

local ok, msg = checkStatus()
if not ok then kick(msg) return end

spawn(function()
    while true do wait(10)
        local still, msg2 = checkStatus()
        if not still then kick(msg2) break end
    end
end)

` + source;
            currentAiCode = bindingCode;
            currentAiScriptId = id;
            document.getElementById('ai-code-output').value = bindingCode;
            document.getElementById('ai-result-box').classList.remove('hidden');
            alert("✅ تم ربط السكريبت بالسيرفر!");
        }

        async function loadMyScripts() {
            try {
                const res = await apiFetch(`${SERVER_URL}/api/list`, { method: 'POST' });
                const scripts = await res.json();
                const container = document.getElementById('files-list');
                const selectDD = document.getElementById('key-script-select');
                container.innerHTML = ""; selectDD.innerHTML = "<option value=''>-- اختر --</option>";
                for (let id in scripts) {
                    const sc = scripts[id];
                    if (sc.type === 'paid') {
                        let opt = document.createElement('option'); opt.value = id; opt.innerText = id;
                        selectDD.appendChild(opt);
                    }
                    container.innerHTML += `
                    <div class="p-3 bg-[#1a0a14]/50 rounded-xl border border-pink-700/30">
                        <div class="flex justify-between">
                            <span class="font-mono text-xs">${id}.lua</span>
                            <span class="text-[10px] ${sc.active ? 'text-green-400' : 'text-red-400'}">${sc.active ? 'شغال' : 'طافي'}</span>
                        </div>
                        <div class="grid grid-cols-4 gap-1 mt-2">
                            <button onclick="toggleScript('${id}', true)" class="bg-green-600 text-[9px] p-1 rounded">تشغيل</button>
                            <button onclick="toggleScript('${id}', false)" class="bg-amber-600 text-[9px] p-1 rounded">إيقاف</button>
                            <button onclick="showLink('${id}','${sc.type}')" class="bg-indigo-600 text-[9px] p-1 rounded">رابط</button>
                            <button onclick="deleteScript('${id}')" class="bg-red-600 text-[9px] p-1 rounded">🗑️</button>
                        </div>
                    </div>`;
                }
            } catch (e) {
                console.error("خطأ في تحميل السكريبتات:", e);
            }
        }

        async function toggleScript(id, status) {

            if (confirm(`⚠️ حذف السكريبت ${id}؟`)) {

                loadMyScripts(); loadMyKeys();
            }
        }
        function showLink(id, type) {
            let ls = `loadstring(game:HttpGet("${SERVER_URL}/load/${id}.lua"))()`;

        async function generateKey() {
            const scriptId = document.getElementById('key-script-select').value;
            const duration = document.getElementById('key-duration').value.trim();
            if (!scriptId) return alert("اختر سكريبت!");
            const res = await apiFetch(`${SERVER_URL}/api/keys/generate`, {
                method: 'POST', body: JSON.stringify({ scriptId, duration })
            });
            const data = await res.json();
            if (data.success) {
                document.getElementById('generated-key-text').innerText = data.key;
                document.getElementById('generated-key-box').classList.remove('hidden');
                loadMyKeys();
            } else alert("❌ " + data.message);
        }

        async function loadMyKeys() {
            try {
                const res = await apiFetch(`${SERVER_URL}/api/keys/list`, { method: 'POST' });
                allKeysData = await res.json();
                filterKeys();
            } catch (e) {
                console.error("خطأ في تحميل المفاتيح:", e);
            }
        }

        function filterKeys() {
            const searchTerm = document.getElementById('search-input').value.toLowerCase();
            const filterType = document.getElementById('search-filter').value;
            let filtered = [...allKeysData];
            if (searchTerm) {
                filtered = filtered.filter(k => {
                    if (filterType === 'key') return k.key.toLowerCase().includes(searchTerm);
                    if (filterType === 'script') return k.scriptId.toLowerCase().includes(searchTerm);
                    if (filterType === 'username') return (k.activatedUsername || '').toLowerCase().includes(searchTerm);
                    return k.key.toLowerCase().includes(searchTerm) || k.scriptId.toLowerCase().includes(searchTerm);
                });
            }
            const tbody = document.getElementById('keys-list-tbody');
            tbody.innerHTML = '';
            filtered.forEach(k => {
                const isExpired = k.isExpired;
                const rowClass = isExpired ? 'expired' : 'active-key';
                tbody.innerHTML += `
                <tr class="${rowClass} key-row">
                    <td class="py-2 font-mono text-yellow-400 text-[11px]">${k.key}</td>
                    <td class="py-2 text-xs">${k.scriptId}.lua</td>
                    <td class="py-2"><span class="duration-badge">${k.duration || 'غير محدد'}</span></td>
                    <td class="py-2"><span class="${isExpired ? 'text-red-400' : 'text-green-400'}">${isExpired ? 'منتهي' : 'شغال'}</span></td>
                    <td class="py-2">${k.activatedUsername || '⚪'}</td>
                    <td class="py-2 text-center">
                        <button onclick="resetKeyHWID('${k.key}')" class="bg-teal-600/30 hover:bg-teal-600 px-2 py-1 rounded text-[10px]">🔄</button>
                        <button onclick="deleteKey('${k.key}')" class="bg-red-600/30 hover:bg-red-600 px-2 py-1 rounded text-[10px]">🗑️</button>
                    </td>
                </tr>`;
            });
        }

        async function resetKeyHWID(key) {
            await apiFetch(`${SERVER_URL}/api/keys/reset`, { method: 'POST', body: JSON.stringify({ key }) });
            loadMyKeys();
        }
        async function deleteKey(key) {
