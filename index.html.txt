    <script>
        const SERVER_URL = window.location.origin;
        let currentAiCode = "";
        let currentAiScriptId = "";
        let allKeysData = [];
        let refreshInterval = null;
        let currentEditId = null;

        // ========== دوال مساعدة - بدون توثيق ==========
        function getHeaders() {
            return {
                'Content-Type': 'application/json'
            };
        }

        async function apiFetch(url, options = {}) {
            const res = await fetch(url, {
                ...options,
                headers: getHeaders()
            });
            if (!res.ok) {
                console.warn("⚠️ استجابة غير متوقعة:", res.status);
            }
            return res;
        }

        function closeResultBox() {
            document.getElementById('result-box').classList.add('hidden');
        }

        function closeAiBox() {
            document.getElementById('ai-result-box').classList.add('hidden');
        }

        function cancelEdit() {
            currentEditId = null;
            document.getElementById('script-id').value = "";
            document.getElementById('source-code').value = "";
            document.getElementById('edit-warning').classList.add('hidden');
            document.getElementById('script-id').classList.remove('edit-mode');
            document.querySelector('input[name="script-type"][value="free"]').checked = true;
        }

        async function refreshKeys() {
            const btn = document.getElementById('refresh-btn');
            btn.classList.add('refresh-animation');
            await loadMyKeys();
            setTimeout(() => btn.classList.remove('refresh-animation'), 500);
        }

        // ========== رفع السكريبت العادي ==========
        async function uploadScript(overrideSource = null, overrideId = null) {
            const id = overrideId || document.getElementById('script-id').value.trim();
            let source = overrideSource || document.getElementById('source-code').value.trim();
            const type = document.querySelector('input[name="script-type"]:checked').value;

            if (!id || !source) return alert("⚠️ املأ الحقول!");

            const res = await apiFetch(`${SERVER_URL}/api/upload`, {
                method: 'POST',
                body: JSON.stringify({ id, source, type })
            });
            const data = await res.json();

            if (data.success) {
                let ls = `loadstring(game:HttpGet("${SERVER_URL}/load/${id}.lua"))()`;
                if (type === 'paid') ls = `_G.Key = "مفتاحك"\n` + ls;
                document.getElementById('final-link-input').value = ls;
                document.getElementById('result-box').classList.remove('hidden');
                if (!overrideSource) {
                    cancelEdit();
                }
                loadMyScripts();
                loadMyKeys();
                alert("✅ تم الحفظ بنجاح!");
            } else alert("❌ " + data.message);
        }

        async function uploadWithKeySystem() {
            const id = document.getElementById('script-id').value.trim();
            let source = document.getElementById('source-code').value.trim();
            const type = document.querySelector('input[name="script-type"]:checked').value;

            if (!id || !source) return alert("⚠️ املأ الحقول!");

            if (type !== 'paid') {
                if (!confirm("⚠️ هذا السكريبت ليس مدفوعاً. هل تريد تحويله إلى مدفوع مع Key System؟")) {
                    return;
                }
                document.querySelector('input[name="script-type"][value="paid"]').checked = true;
            }

            const res = await apiFetch(`${SERVER_URL}/api/upload`, {
                method: 'POST',
                body: JSON.stringify({ id, source, type: 'paid' })
            });
            const data = await res.json();

            if (data.success) {
                let ls = `loadstring(game:HttpGet("${SERVER_URL}/load/${id}.lua"))()`;
                ls = `_G.Key = "مفتاحك"\n` + ls;
                document.getElementById('final-link-input').value = ls;
                document.getElementById('result-box').classList.remove('hidden');
                cancelEdit();
                loadMyScripts();
                loadMyKeys();
                alert("✅ تم رفع الكود مع Key System بنجاح!");
            } else {
                alert("❌ فشل الرفع: " + data.message);
            }
        }

        async function injectKeyWithAI() {
            const id = document.getElementById('script-id').value.trim();
            const source = document.getElementById('source-code').value.trim();
            const type = document.querySelector('input[name="script-type"]:checked').value;
            if (!id || !source) return alert("⚠️ املأ الحقول!");
            if (type !== 'paid') return alert("⚠️ اختر مدفوع!");
            const btn = document.getElementById('ai-btn-text');
            btn.innerHTML = '<span class="ai-loading">جاري...</span>';
            const res = await apiFetch(`${SERVER_URL}/api/ai/inject-key`, {
                method: 'POST',
                body: JSON.stringify({ source, scriptId: id, serverUrl: SERVER_URL })
            });
            const data = await res.json();
            btn.innerHTML = '🤖 دمج Key System';
            if (data.success) {
                currentAiCode = data.code;
                currentAiScriptId = id;
                document.getElementById('ai-code-output').value = data.code;
                document.getElementById('ai-result-box').classList.remove('hidden');
            } else alert("❌ خطأ");
        }

        // ========== ربط السكريبت بالسيرفر (حماية ضد النسخ) ==========
        async function injectServerBinding() {
            const id = document.getElementById('script-id').value.trim();
            let source = document.getElementById('source-code').value.trim();

            if (!id || !source) {
                return alert("⚠️ اكتب اسم الملف وضع الكود أولاً!");
            }

            const btn = event.target;
            const originalText = btn.innerHTML;
            btn.innerHTML = '<span class="ai-loading">🔗 جاري الربط مع السيرفر...</span>';
            btn.disabled = true;

            const bindingCode = `-- ============================================
        -- 🔗 KVN Server Binding System
        -- ربط السكريبت بالسيرفر - منع النسخ
        -- لو أوقفته من الموقع => يتقفل على الكل فوراً
        -- ============================================

        local Players = game:GetService("Players")
        local StarterGui = game:GetService("StarterGui")
        local HttpService = game:GetService("HttpService")

        local SERVER_URL = "${SERVER_URL}"
        local SCRIPT_ID = "${id}"
        local localPlayer = Players.LocalPlayer

        -- التحقق من حالة السكريبت من السيرفر
        local function checkServerStatus()
            local url = SERVER_URL .. "/api/script-status/" .. SCRIPT_ID

            local success, response = pcall(function()
                return game:HttpGet(url)
            end)

            if not success then
                return false, "فشل الاتصال بالسيرفر"
            end

            local decoded
            local decodeSuccess = pcall(function()
                decoded = HttpService:JSONDecode(response)
            end)

            if not decodeSuccess or not decoded then
                return false, "خطأ في استجابة السيرفر"
            end

            if not decoded.active then
                return false, decoded.message or "⛔ السكريبت متوقف من قبل الأدمن"
            end

            return true, "✅ السكريبت شغال"
        end

        -- طرد اللاعب
        local function kickPlayer(reason)
            pcall(function()
                StarterGui:SetCore("SendNotification", {
                    Title = "🔒 تم إيقاف السكريبت",
                    Text = reason or "تم إيقاف هذا السكريبت من لوحة التحكم",
                    Duration = 5
                })
            end)
            wait(1)
            pcall(function()
                localPlayer:Kick(reason or "⛔ السكريبت متوقف")
            end)
        end

        -- التحقق المستمر
        local function startServerBinding()
            local isActive, msg = checkServerStatus()

            if not isActive then
                kickPlayer(msg)
                return false
            end

            pcall(function()
                StarterGui:SetCore("SendNotification", {
                    Title = "🔗 السكريبت مربوط بالسيرفر",
                    Text = "✅ إذا أوقفه الأدمن سيتوقف عندك فوراً",
                    Duration = 3
                })
            end)

            -- كل 10 ثواني يتحقق
            spawn(function()
                while true do
                    wait(10)
                    local stillActive, kickMsg = checkServerStatus()
                    if not stillActive then
                        kickPlayer(kickMsg)
                        break
                    end
                end
            end)

            return true
        end

        if not startServerBinding() then
            return
        end

        -- ============================================
        -- كودك الأصلي يبدأ من هنا
        -- ============================================

        ` + source;

            currentAiCode = bindingCode;
            currentAiScriptId = id;
            document.getElementById('ai-code-output').value = bindingCode;
            document.getElementById('ai-result-box').classList.remove('hidden');

            btn.innerHTML = originalText;

        // ========== تحميل قائمة السكريبتات ==========
        async function loadMyScripts() {
            try {
                const res = await apiFetch(`${SERVER_URL}/api/list`, { method: 'POST' });
                const scripts = await res.json();
                const container = document.getElementById('files-list');
                const selectDD = document.getElementById('key-script-select');
                container.innerHTML = "";
                selectDD.innerHTML = "<option value=''>-- اختر --</option>";

                for (let id in scripts) {
                    const sc = scripts[id];
                    if (sc.type === 'paid') {
                        let opt = document.createElement('option');
                        opt.value = id;
                        opt.innerText = id;
                        selectDD.appendChild(opt);
                    }
                    container.innerHTML += `
                <div class="p-3 bg-[#1f2937]/50 rounded-xl border border-gray-700">
                    <div class="flex justify-between">
                        <span class="font-mono text-xs">${id}.lua</span>
                        <span class="text-[10px] ${sc.active ? 'text-green-400' : 'text-red-400'}">${sc.active ? 'شغال' : 'طافي'}</span>
                    </div>
                    <div class="grid grid-cols-4 gap-1 mt-2">
                        <button onclick="toggleScript('${id}', true)" class="bg-green-600 text-[9px] p-1 rounded">تشغيل</button>
                        <button onclick="toggleScript('${id}', false)" class="bg-amber-600 text-[9px] p-1 rounded">إيقاف</button>
                        <button onclick="showLink('${id}','${sc.type}')" class="bg-indigo-600 text-[9px] p-1 rounded">رابط</button>
                        <button onclick="editScript('${id}')" class="bg-blue-600 text-[9px] p-1 rounded">✏️ تعديل</button>
                        <button onclick="deleteScript('${id}')" class="bg-red-600 text-[9px] p-1 rounded">🗑️ حذف</button>
                    </div>
                </div>`;
                }
            } catch (e) {
                console.error("خطأ في تحميل السكريبتات:", e);
            }
        }

        async function editScript(id) {
            const res = await apiFetch(`${SERVER_URL}/api/get-script-source`, {
                method: 'POST',
                body: JSON.stringify({ id })
            });
            const data = await res.json();

            if (data.success) {
                currentEditId = id;
                document.getElementById('source-code').value = data.source;
                document.getElementById('script-id').value = id;
                document.getElementById('script-id').classList.add('edit-mode');
                document.getElementById('edit-warning').classList.remove('hidden');

                const res2 = await apiFetch(`${SERVER_URL}/api/list`, { method: 'POST' });
                const scripts = await res2.json();

                if (scripts[id] && scripts[id].type === 'paid') {
                    document.querySelector('input[name="script-type"][value="paid"]').checked = true;
                } else {
                    document.querySelector('input[name="script-type"][value="free"]').checked = true;
                }

                window.scrollTo({ top: 0, behavior: 'smooth' });
                alert(`✏️ جاري تعديل: ${id}.lua\nبعد التعديل اضغط "حفظ وإنشاء رابط"`);
            } else {
                alert("❌ خطأ في جلب الكود: " + data.message);
            }
        }

        // ========== إدارة المفاتيح ==========
        async function generateKey() {
            const scriptId = document.getElementById('key-script-select').value;
            const duration = document.getElementById('key-duration').value.trim();
            if (!scriptId) return alert("اختر سكريبت!");
            const res = await apiFetch(`${SERVER_URL}/api/keys/generate`, {
                method: 'POST',
                body: JSON.stringify({ scriptId, duration })
            });
            const data = await res.json();
            if (data.success) {
                document.getElementById('generated-key-text').innerText = data.key;
                document.getElementById('generated-expiry').innerHTML = data.expiryDate ?
                    `⏰ تنتهي: ${new Date(data.expiryDate).toLocaleString('ar')}` : '⏰ غير محددة';
                document.getElementById('generated-key-box').classList.remove('hidden');
                loadMyKeys();
                setTimeout(() => document.getElementById('generated-key-box').classList.add('hidden'), 10000);
            } else alert("❌ " + data.message);
        }

        async function loadMyKeys(silent = false) {
            try {
                const res = await apiFetch(`${SERVER_URL}/api/keys/list`, { method: 'POST' });
                allKeysData = await res.json();
                filterKeys();
            } catch (e) {
                console.error("خطأ في تحميل المفاتيح:", e);
            }
        }

        async function updateKeyDuration(key) {
            const newDuration = prompt("أدخل المدة الجديدة:\nمثال: 1h, 2d, 3m, 1y, 30min");
            if (!newDuration) return;
            const res = await apiFetch(`${SERVER_URL}/api/keys/update-duration`, {
                method: 'POST',
                body: JSON.stringify({ key, duration: newDuration })
            });
            const data = await res.json();
            if (data.success) {
                alert(`✅ تم التحديث!`);
                loadMyKeys();
            } else alert("❌ " + data.message);
        }

        function filterKeys() {
            const searchTerm = document.getElementById('search-input').value.toLowerCase();
            const filterType = document.getElementById('search-filter').value;
            let filtered = [...allKeysData];

            if (searchTerm) {
                filtered = filtered.filter(k => {
                    if (filterType === 'key') return k.key.toLowerCase().includes(searchTerm);
                    if (filterType === 'script') return k.scriptId.toLowerCase().includes(searchTerm);
                    if (filterType === 'username') return (k.activatedUsername || '').toLowerCase()
                        .includes(searchTerm);
                    return k.key.toLowerCase().includes(searchTerm) ||
                        k.scriptId.toLowerCase().includes(searchTerm) ||
                        (k.activatedUsername || '').toLowerCase().includes(searchTerm);
                });
            }

            document.getElementById('search-count').innerHTML =
                `📊 ${filtered.length}/${allKeysData.length}`;
            const tbody = document.getElementById('keys-list-tbody');
            const noResults = document.getElementById('no-results-message');

            if (filtered.length === 0) {
                tbody.innerHTML = '';
                noResults.classList.remove('hidden');
                return;
            }

            noResults.classList.add('hidden');
            tbody.innerHTML = '';

            filtered.forEach(k => {
                const isExpired = k.isExpired;
                const rowClass = isExpired ? 'expired-key' : 'active-key';
                const expiryDate = k.expiryDate ? new Date(k.expiryDate).toLocaleString('ar') :
                    'غير محددة';
                const remaining = k.remaining || (isExpired ? 'منتهي ❌' : 'غير محددة');
                const boundUser = k.activatedUsername || '⚪ غير مرتبط';

                tbody.innerHTML += `
            <tr class="${rowClass} border-b border-gray-800">
                <td class="py-2 font-mono text-yellow-400 text-[11px]">${k.key}</td>
                <td class="py-2 font-mono text-xs">${k.scriptId}.lua</td>
                <td class="py-2"><span class="duration-badge">${k.duration || 'غير محدد'}</span></td>
                <td class="py-2"><span class="remaining-badge ${isExpired ? 'text-red-400' : 'text-green-400'}">${remaining}</span><br><span class="text-[9px] text-gray-500">${expiryDate}</span></td>
                <td class="py-2"><span class="${k.activatedUsername ? 'text-green-400 font-bold' : 'text-gray-500'}">${boundUser}</span></td>
                <td class="py-2 text-center">
                    <button onclick="updateKeyDuration('${k.key}')" class="bg-blue-600/30 hover:bg-blue-600 text-white px-2 py-1 rounded text-[10px] mx-1">✏️ تعديل</button>
                    <button onclick="resetKeyHWID('${k.key}')" class="bg-teal-600/30 hover:bg-teal-600 px-2 py-1 rounded text-[10px] mx-1">🔄 تصفير</button>
                    <button onclick="deleteKey('${k.key}')" class="bg-red-600/30 hover:bg-red-600 px-2 py-1 rounded text-[10px] mx-1">🗑️ حذف</button>
                </td>
            </tr>`;
            });
        }
